rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function memberDoc(churchId) {
      return get(/databases/$(database)/documents/churches/$(churchId)/members/$(request.auth.uid));
    }

    function role(churchId) {
      return signedIn() && memberDoc(churchId).data.role;
    }

    function isAdmin(churchId) {
      return signedIn() && (
        role(churchId) == 'admin' ||
        role(churchId) == 'pastor'
      );
    }

    match /churches/{churchId} {
      allow read: if signedIn();

      match /members/{memberId} {
        allow read: if signedIn();
        allow create: if signedIn() && memberId == request.auth.uid;
        allow update: if isAdmin(churchId) || memberId == request.auth.uid;
        allow delete: if isAdmin(churchId);
      }

      match /announcements/{announcementId} {
        allow read: if signedIn();
        allow write: if isAdmin(churchId);
      }

      match /events/{eventId} {
        allow read: if signedIn();
        allow write: if isAdmin(churchId);
      }

      match /eventCheckIns/{checkInId} {
        allow read: if signedIn();
        allow create: if signedIn();
        allow update, delete: if isAdmin(churchId);
      }

      match /media/{mediaId} {
        allow read: if signedIn();
        allow write: if isAdmin(churchId);
      }

      match /ministries/{ministryId} {
        allow read: if signedIn();
        allow write: if isAdmin(churchId);
      }

      match /small_groups/{groupId} {
        allow read: if signedIn();
        allow write: if isAdmin(churchId);
      }

      match /volunteer_assignments/{assignmentId} {
        allow read: if signedIn();
        allow write: if isAdmin(churchId);
      }
    }
  }
}
